## 4. 조건문/반복문

## javascript로 워들 게임 만들어보기
```jsx
<body>

    <style>
        input {
            width: 50px;
            height: 50px;
            font-size: 40px;
            text-align: center;
        }
    </style>

    <input class="input">
    <input class="input">
    <input class="input">
    <input class="input">
    <input class="input">

    <button>제출</button>

    <script>
        //제출 누르면 답 검사
        var answer = 'abcde';
        document.querySelector('button').addEventListener('click',
            //제출버튼 누르면 함수 실행
        function () {

                //사용자가 입력한 값 input이라는 변수로 가져옴
                var input = document.querySelectorAll('.input');
                for (let i = 0; i < 5; i++) {
                    //1. 위치까지 맞으면 -> 초록
                    if (input[i].value == answer[i]) {
                        input[i].style.background = 'green';
                    }
                    //2. 위치는 안맞음 -> 노란칠
                    else if (answer.includes(input[i].value)) {
                        input[i].style.background = 'yellow';
                    }
                    //3. 그것들 다 아니면-> 회색칠
                    else {
                        input[i].style.background = 'lightgrey';
                    }
                }
                //제출버튼 누르면 input 5개도 새로 생성
                var template = `<div>
                    <input class="input">
                    <input class="input">
                    <input class="input">
                    <input class="input">
                    <input class="input">
                </div>`;
                document.querySelector('body').insertAdjacentHTML
                ('beforeend', template); //body에 추가한다는 문법


            }) 
    </script>

</body>
```
